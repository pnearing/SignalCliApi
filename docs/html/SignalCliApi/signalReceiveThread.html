 29def __type_check_callback__(callback: Optional[tuple[Callable, Optional[list[Any]]]]) -> bool:
 30    """
 31    Type check a call back.
 32    :param callback: The call back to type check.
 33    :return: bool: True or False on success or failure.
 34    """
 35    if callback is not None:
 36        if not isinstance(callback, tuple):
 37            return False
 38        elif len(callback) != 2:
 39            return False
 40        elif not callable(callback[CallbackIdx.CALLABLE]):
 41            return False
 42        elif callback[CallbackIdx.PARAMS] is not None and not isinstance(callback[CallbackIdx.PARAMS], list):
 43            return False
 44    return True

196    def __call_callback__(self,
197                          callback: Optional[tuple[Callable, Optional[list[Any]]]],
198                          account: Account,
199                          message: Message,
200                          ) -> Optional[bool]:
201        """
202        Execute a callback and return True for stopping reception, False for do not stop reception; The order of
203        priority of return values is all messages callback, specified callback. So if all messages returns None, then
204        the return value of the specified callback is returned.
205        :param callback: Optional[tuple[Callable, Optional[list[Any]]]]: The callback to call, and any parameters to
206            pass to it, if None the callback is not executed.
207        :param account: Account: The account we're receiving for.
208        :param message: Message: The message we've received.
209        :return: Optional[bool]: If True is returned the callback stops the reception thread, If False or None are
210            returned then reception continues.
211        """
212        # Setup logging:
213        logger: logging.Logger = logging.getLogger(__name__ + '.' + self.__call_callback__.__name__)
214
215        # Return if callback is None:
216        cb_return_value: Optional[bool] = None
217        # Call specified call back:
218        try:
219            logger.debug("Calling callback: '%s'" % callback[CallbackIdx.CALLABLE].__name__)
220            if callback[CallbackIdx.PARAMS] is not None:
221                cb_return_value = callback[CallbackIdx.CALLABLE](account, message, *callback[CallbackIdx.PARAMS])
222            else:
223                cb_return_value = callback[CallbackIdx.CALLABLE](account, message)
224        except Exception as e:
225            error_message: str = "Callback '%s', caused an Exception '%s', with arguments: '%s'" \
226                                 % (callback[CallbackIdx.CALLABLE].__name__, str(type(e)), str(e.args))
227            logger.critical(error_message)
228
229        # Call all messages callback:
230        all_return_value: Optional[bool] = None
231        try:
232            logger.debug("Calling callback: '%s'." % self._all_msg_cb[CallbackIdx.CALLABLE].__name__)
233            if self._all_msg_cb[CallbackIdx.PARAMS] is not None:
234                all_return_value = self._all_msg_cb[CallbackIdx.CALLABLE](account, message,
235                                                                          *self._all_msg_cb[CallbackIdx.PARAMS])
236            else:
237                all_return_value = self._all_msg_cb[CallbackIdx.CALLABLE](account, message)
238        except Exception as e:
239            error_message: str = "Callback '%s', caused an Exception '%s', with arguments: '%s'" \
240                                 % (callback[CallbackIdx.CALLABLE].__name__, str(type(e)), str(e.args))
241            logger.critical(error_message)
242
243        # Determine return value:
244        if all_return_value is not None:
245            return all_return_value
246        return cb_return_value

248    def __parse_data_message__(self, envelope_dict: dict[str, Any]) -> Optional[bool]:
249        """
250        Parse a dataMessage incoming message.
251        :param envelope_dict: dict[str, Any]: The dict provided by signal.
252        :return: Optional[bool]: The return value of the callbacks.
253        """
254        # Setup logging:
255        logger: logging.Logger = logging.getLogger(__name__ + '.' + self.__parse_data_message__.__name__)
256
257        # Fetch data message:
258        data_message: dict[str, Any] = envelope_dict['dataMessage']
259
260        #######################
261        # REACTIONS:
262        #######################
263        if 'reaction' in data_message.keys():
264            # Create reaction Message:
265            message = Reaction(
266                command_socket=self._command_socket, account_id=self._account.number,
267                config_path=self._config_path, contacts=self._account.contacts,
268                groups=self._account.groups, devices=self._account.devices,
269                this_device=self._account.device, raw_message=envelope_dict
270            )
271            # Parse reaction and call the reaction callback:
272            logger.debug("Got reaction message, parsing and calling reaction callback.")
273            self._account.messages.__parse_reaction__(message)
274            message.sender.seen(message.timestamp)
275            return self.__call_callback__(self._ract_msg_cb, self._account, message)
276        ######################
277        # GROUP UPDATES:
278        ######################
279        else:
280            # TODO: See if there is a better way to do this, this feels and reads pretty hacky.
281            is_group_update: bool
282            try:
283                if data_message['groupInfo']['type'] == 'UPDATE':
284                    is_group_update = True
285                else:
286                    is_group_update = False
287            except KeyError:
288                is_group_update = False
289            if is_group_update:
290                message = GroupUpdate(
291                    command_socket=self._command_socket, account_id=self._account.number,
292                    config_path=self._config_path, contacts=self._account.contacts,
293                    groups=self._account.groups, devices=self._account.devices,
294                    this_device=self._account.device, raw_message=envelope_dict
295                )
296                logger.debug("Got a group update message, syncing groups and calling sync callback.")
297                message.recipient.__sync__()
298                self._account.messages.append(message)
299                message.sender.seen(message.timestamp)
300                return self.__call_callback__(self._sync_msg_cb, self._account, message)
301            ##################################
302            # Received Message:
303            ##################################
304            else:
305                # Create a Received message:
306                message = ReceivedMessage(
307                    command_socket=self._command_socket, account_id=self._account.number,
308                    config_path=self._config_path, contacts=self._account.contacts,
309                    groups=self._account.groups, devices=self._account.devices,
310                    this_device=self._account.device, sticker_packs=self._sticker_packs,
311                    raw_message=envelope_dict,
312                )
313                logger.debug("Got a received message, storing and calling received callback.")
314                # Sender is no longer typing:
315                if message.sender.is_typing:
316                    # Create a typing stopped message:
317                    stop_typing_message = TypingMessage(command_socket=self._command_socket,
318                                                        account_id=self._account.get_id(), config_path=self._config_path,
319                                                        contacts=self._account.contacts, groups=self._account.groups,
320                                                        devices=self._account.devices, this_device=self._account.device,
321                                                        sender=message.sender, recipient=message.recipient,
322                                                        device=message.device, timestamp=message.timestamp,
323                                                        action=TypingStates.STOPPED, time_changed=message.timestamp)
324                    # Send the typing message to the sending for parsing:
325                    message.sender.__parse_typing_message__(stop_typing_message)
326                    # Store the typing message.
327                    self._account.messages.append(stop_typing_message)
328                # Store the received message:
329                self._account.messages.append(message)
330                # Mark the sender as seen:
331                message.sender.seen(message.timestamp)
332                return self.__call_callback__(self._recv_msg_cb, self._account, message)

351    def __parse_sync_message__(self, envelope_dict: dict[str, Any]) -> Optional[bool]:
352        """
353        Parse a sync message.
354        :param envelope_dict: dict[str, Any]: The incoming message.
355        :return: Optional[bool]: The return value of the callback.
356        """
357        # Setup logging:
358        logger: logging.Logger = logging.getLogger(__name__ + '.' + self.__parse_sync_message__.__name__)
359
360        # Check to see if this is an empty sync message:
361        if envelope_dict['sync_message'] == {}:
362            logger.warning("Got empty sync message, skipping.")
363            return None
364        # Create the SyncMessage object:
365        message = SyncMessage(
366            command_socket=self._command_socket, account_id=self._account.number,
367            config_path=self._config_path, contacts=self._account.contacts,
368            groups=self._account.groups, devices=self._account.devices,
369            this_device=self._account.device, sticker_packs=self._sticker_packs,
370            raw_message=envelope_dict
371        )
372        if message.sync_type == SyncTypes.READ_MESSAGES or message.sync_type == SyncTypes.SENT_MESSAGES:
373            self._account.messages.__parse_sync_message__(message)
374        elif message.sync_type == SyncTypes.CONTACTS:
375            self._account.contacts.__sync__()
376        elif message.sync_type == SyncTypes.GROUPS:
377            self._account.groups.__sync__()
378        elif message.sync_type == SyncTypes.BLOCKS:
379            self._account.contacts.__parse_sync_message__(message)
380            self._account.groups.__parse_sync_message__(message)
381        elif message.sync_type == SyncTypes.SENT_MESSAGES:
382            self._account.messages.__parse_sync_message__(message)
383        else:
384            error_message: str = "Unhandled sync type: %s" % str(message.sync_type)
385            logger.critical("Raising RuntimeError(%s)." % error_message)
386            raise RuntimeError(error_message)
387        # Append the message to messages:
388        self._account.messages.append(message)
389        # Call sync message callback:
390        return self.__call_callback__(self._sync_msg_cb, self._account, message)

438    def run(self) -> None:
439        """
440        Thread override.
441        :return: None
442        """
443        # Setup logging:
444        logger: logging.Logger = logging.getLogger(__name__ + '.' + self.run.__name__)
445
446        # Do send sync request if we're not the primary device.
447        if self._account.device_id != 1:
448            # Create sync request object and json command string:
449            sync_request_command_obj: dict[str, Any] = {
450                "jsonrpc": "2.0",
451                "id": 10,
452                "method": "sendSyncRequest",
453                "params": {
454                    "account": self._account.number,
455                }
456            }
457            json_command_str: str = json.dumps(sync_request_command_obj) + '\n'
458
459            # Communicate with Signal:
460            __socket_send__(self._receive_socket, json_command_str)
461            response_str = __socket_receive__(self._receive_socket)
462            response_obj: dict[str, Any] = __parse_signal_response__(response_str)
463
464            error_occurred, signal_code, signal_message = __check_response_for_error__(response_obj, [])
465            if error_occurred:
466                error_message: str = "signal error while sending sync request. Code %i, Message: %s" \
467                                     % (signal_code, signal_message)
468                logger.warning(error_message)
469
470        # Create receive object and json command string:
471        start_receive_command_object = {
472            "jsonrpc": "2.0",
473            "id": 1,
474            "method": "subscribeReceive",
475            "params": {
476                "account": self._account.number,
477            }
478        }
479        json_command_str: str = json.dumps(start_receive_command_object) + '\n'
480
481        # Communicate start receive with signal:
482        __socket_send__(self._receive_socket, json_command_str)
483        response_str = __socket_receive__(self._receive_socket)
484        response_obj: dict[str, Any] = __parse_signal_response__(response_str)
485        error_occurred, signal_code, signal_message = __check_response_for_error__(response_obj, [])
486        if error_occurred:
487            error_message: str = "Signal error while trying to start receiving. Code %i, Message: %s" \
488                                 % (signal_code, signal_message)
489            logger.critical("Raising RuntimeError(%s)." % error_message)
490            raise RuntimeError(error_message)
491
492        # Set subscription ID.
493        self._subscription_id = response_obj['result']
494
495        # START RECEIVE LOOP:
496        while self._subscription_id is not None:
497            # Blocks until a message received:
498            try:
499                response_str: str = __socket_receive__(self._receive_socket)
500            except CommunicationsError as e:
501                logger.critical("Failed to receive, got CommunicationsError(%s)." % str(e.args))
502                break
503            # Delay processing until messages are finished sending:
504            if self._account.messages.sending:
505                logger.debug("Message received while sending a message, delaying processing until complete.")
506            while self._account.messages.sending:
507                pass
508
509            # Parse incoming message:
510            message_obj: dict[str, Any] = __parse_signal_response__(response_str)
511            if 'method' in message_obj.keys() and message_obj['method'] == 'receive':
512                envelope_dict: dict = message_obj['params']['envelope']
513                return_value: Optional[bool] = None
514                ##########################
515                # Data Message:
516                ##########################
517                if 'dataMessage' in envelope_dict.keys():
518                    return_value = self.__parse_data_message__(envelope_dict)
519                    if return_value is True:
520                        break  # Stop receiving.
521                ##########################
522                # Receipt Message:

197    def __call_callback__(self,
198                          callback: Optional[tuple[Callable, Optional[list[Any]]]],
199                          account: Account,
200                          message: Message,
201                          ) -> Optional[bool]:
202        """
203        Execute a callback and return True for stopping reception, False for do not stop reception; The order of
204        priority of return values is all messages callback, specified callback. So if all messages returns None, then
205        the return value of the specified callback is returned.
206        :param callback: Optional[tuple[Callable, Optional[list[Any]]]]: The callback to call, and any parameters to
207            pass to it, if None the callback is not executed.
208        :param account: Account: The account we're receiving for.
209        :param message: Message: The message we've received.
210        :return: Optional[bool]: If True is returned the callback stops the reception thread, If False or None are
211            returned then reception continues.
212        """
213        # Setup logging:
214        logger: logging.Logger = logging.getLogger(__name__ + '.' + self.__call_callback__.__name__)
215
216        # Return if callback is None:
217        cb_return_value: Optional[bool] = None
218        # Call specified call back:
219        try:
220            logger.debug("Calling callback: '%s'" % callback[CallbackIdx.CALLABLE].__name__)
221            if callback[CallbackIdx.PARAMS] is not None:
222                cb_return_value = callback[CallbackIdx.CALLABLE](account, message, *callback[CallbackIdx.PARAMS])
223            else:
224                cb_return_value = callback[CallbackIdx.CALLABLE](account, message)
225        except Exception as e:
226            error_message: str = "Callback '%s', caused an Exception '%s', with arguments: '%s'" \
227                                 % (callback[CallbackIdx.CALLABLE].__name__, str(type(e)), str(e.args))
228            logger.critical(error_message)
229
230        # Call all messages callback:
231        all_return_value: Optional[bool] = None
232        try:
233            logger.debug("Calling callback: '%s'." % self._all_msg_cb[CallbackIdx.CALLABLE].__name__)
234            if self._all_msg_cb[CallbackIdx.PARAMS] is not None:
235                all_return_value = self._all_msg_cb[CallbackIdx.CALLABLE](account, message,
236                                                                          *self._all_msg_cb[CallbackIdx.PARAMS])
237            else:
238                all_return_value = self._all_msg_cb[CallbackIdx.CALLABLE](account, message)
239        except Exception as e:
240            error_message: str = "Callback '%s', caused an Exception '%s', with arguments: '%s'" \
241                                 % (callback[CallbackIdx.CALLABLE].__name__, str(type(e)), str(e.args))
242            logger.critical(error_message)
243
244        # Determine return value:
245        if all_return_value is not None:
246            return all_return_value
247        return cb_return_value

439    def run(self) -> None:
440        """
441        Thread override.
442        :return: None
443        """
444        # Setup logging:
445        logger: logging.Logger = logging.getLogger(__name__ + '.' + self.run.__name__)
446
447        # Do send sync request if we're not the primary device.
448        if self._account.device_id != 1:
449            # Create sync request object and json command string:
450            sync_request_command_obj: dict[str, Any] = {
451                "jsonrpc": "2.0",
452                "id": 10,
453                "method": "sendSyncRequest",
454                "params": {
455                    "account": self._account.number,
456                }
457            }
458            json_command_str: str = json.dumps(sync_request_command_obj) + '\n'
459
460            # Communicate with Signal:
461            __socket_send__(self._receive_socket, json_command_str)
462            response_str = __socket_receive__(self._receive_socket)
463            response_obj: dict[str, Any] = __parse_signal_response__(response_str)
464
465            error_occurred, signal_code, signal_message = __check_response_for_error__(response_obj, [])
466            if error_occurred:
467                error_message: str = "signal error while sending sync request. Code %i, Message: %s" \
468                                     % (signal_code, signal_message)
469                logger.warning(error_message)
470
471        # Create receive object and json command string:
472        start_receive_command_object = {
473            "jsonrpc": "2.0",
474            "id": 1,
475            "method": "subscribeReceive",
476            "params": {
477                "account": self._account.number,
478            }
479        }
480        json_command_str: str = json.dumps(start_receive_command_object) + '\n'
481
482        # Communicate start receive with signal:
483        __socket_send__(self._receive_socket, json_command_str)
484        response_str = __socket_receive__(self._receive_socket)
485        response_obj: dict[str, Any] = __parse_signal_response__(response_str)
486        error_occurred, signal_code, signal_message = __check_response_for_error__(response_obj, [])
487        if error_occurred:
488            error_message: str = "Signal error while trying to start receiving. Code %i, Message: %s" \
489                                 % (signal_code, signal_message)
490            logger.critical("Raising RuntimeError(%s)." % error_message)
491            raise RuntimeError(error_message)
492
493        # Set subscription ID.
494        self._subscription_id = response_obj['result']
495
496        # START RECEIVE LOOP:
497        while self._subscription_id is not None:
498            # Blocks until a message received:
499            try:
500                response_str: str = __socket_receive__(self._receive_socket)
501            except CommunicationsError as e:
502                logger.critical("Failed to receive, got CommunicationsError(%s)." % str(e.args))
503                break
504            # Delay processing until messages are finished sending:
505            if self._account.messages.sending:
506                logger.debug("Message received while sending a message, delaying processing until complete.")
507            while self._account.messages.sending:
