201        # Load the dict:

223    def __parse_reaction__(self, reaction: Reaction) -> bool:
224        # Get the messages from the recipients:
225        search_messages: list[Message] = []
226        if reaction.recipient_type == 'contact':
227            messages = self.get_by_sender(reaction.recipient)
228            search_messages.extend(messages)
229        elif reaction.recipient_type == 'group':
230            messages = self.get_by_recipient(reaction.recipient)
231            search_messages.extend(messages)
232        else:
233            # Invalid recipient type:
234            error_message = "Invalid reaction cannot parse."
235            raise RuntimeError(error_message)
236        # Find the message that was reacted to:
237        reactedMessage: Optional[SentMessage | ReceivedMessage | Message] = None
238        for message in search_messages:
239            if message.sender == reaction.target_author:
240                if message.timestamp == reaction.target_timestamp:
241                    reactedMessage = message
242        # If the message isn't in history, do nothing:
243        if reactedMessage is None:
244            return False
245        # Have the message add / change / remove the reaction:
246        reactedMessage.reactions.__parse__(reaction)
247        return True

450                     recipients: Iterable[Contact | Group] | Contact | Group,
451                     body: Optional[str] = None,
452                     attachments: Optional[Iterable[Attachment | str] | Attachment | str] = None,
453                     mentions: Optional[Iterable[Mention] | Mentions | Mention] = None,
454                     quote: Optional[Quote] = None,
455                     sticker: Optional[Sticker] = None,
456                     preview: Optional[Preview] = None,
457                     ) -> tuple[
458        tuple[bool, Contact, str] | tuple[bool, Contact | Group, str] | tuple[bool, Contact, SentMessage] | tuple[
459            bool, Contact, object], ...]:
460        """
461        Send a message.
462        :param recipients: Iterable[Contact | Group] | Contact | Group: The recipients of the message.
463        :param body: Optional[str]: The body of the message.
464        :param attachments: Optional[Iterable[Attachment | str] | Attachment | str ]: Attachments to the message.
465        :param mentions: Optional[Iterable[Mention] | Mentions | Mention]: Mentions in the message.
466        :param quote: Optional[Quote]: A Quote object for the message.
467        :param sticker: Optional[Sticker]: A sticker to send.
468        :param preview: Optional[Preview]: A preview for the url in the message, url must appear in the body of the
469            message.
470        :returns: tuple[tuple[bool, Contact | Group, str | SentMessage]]: True / False for if the message was sent
471            successfully, Contact | Group the message was sent to, str | SentMessage, a string containing an
472            error message or the SentMessage object.
473        :raises: TypeError: If a recipient is not a Contact or Group object, if body is not a string, if attachments is
474            not an Attachment object or a string, or a list of Attachment objects, or strings, if mentions is not a
475            list of Mention objects, or not a Mentions object, if quote is not a Quote object, if sticker is not a
476            Sticker object, or if preview is not a Preview object.
477        :raises: ValueError: If body is an empty string, if attachments is an empty list, or if mentions is an empty
478            list.
479        """
480        # Validate recipients:
481        recipient_type: str = ''
482        target_recipients: list[Contact | Group]
483        if isinstance(recipients, Contact):
484            recipient_type = 'contact'
485            target_recipients = [recipients]
486        elif isinstance(recipients, Group):
487            recipient_type = 'group'
488            target_recipients = [recipients]
489        elif isinstance(recipients, Iterable):
490            target_recipients = []
491            checkType = None
492            for i, recipient in enumerate(recipients):
493                if not isinstance(recipient, Contact) and not isinstance(recipient, Group):
494                    __type_error__("recipients[%i]" % i, "Contact | Group", recipient)
495                if i == 0:
496                    checkType = type(recipient)
497                    if isinstance(recipient, Contact):
498                        recipient_type = 'contact'
499                    else:
500                        recipient_type = 'group'
501                elif not isinstance(recipient, checkType):
502                    __type_error__("recipients[%i]", str(checkType), recipient)
503                target_recipients.append(recipient)
504        else:
505            __type_error__("recipients", "Iterable[Contact | Group] | Contact | Group", recipients)
506        if len(target_recipients) == 0:
507            raise ValueError("recipients cannot be of zero length")
508        # Validate body Type and value:
509        if body is not None and not isinstance(body, str):
510            __type_error__("body", "str | None", body)
511        elif body is not None and len(body) == 0:
512            raise ValueError("body cannot be empty string")
513        # Validate attachments:
514        target_attachments: Optional[list[Attachment]] = None
515        if attachments is not None:
516            if isinstance(attachments, Attachment):
517                target_attachments = [attachments]
518            elif isinstance(attachments, str):
519                target_attachments = [Attachment(config_path=self._config_path, local_path=attachments)]
520            elif isinstance(attachments, Iterable):
521                target_attachments = []
522                for i, attachment in enumerate(attachments):
523                    if not isinstance(attachment, Attachment) and not isinstance(attachment, str):
524                        __type_error__("attachments[%i]" % i, "Attachment | str", attachment)
525                    if isinstance(attachment, Attachment):
526                        target_attachments.append(attachment)
527                    else:
528                        target_attachments.append(Attachment(config_path=self._config_path, local_path=attachment))
529            else:
530                __type_error__("attachments", "Iterable[Attachment | str] | Attachment | str", attachments)
531        if target_attachments is not None and len(target_attachments) == 0:
532            raise ValueError("attachments cannot be empty")
533        # Validate mentions:
534        target_mentions: Optional[list[Mention] | Mentions] = None
535        if mentions is not None:
536            if isinstance(mentions, Mentions):
537                target_mentions = mentions
538            elif isinstance(mentions, Mention):
539                target_mentions = [mentions]
540            elif isinstance(mentions, Iterable):
541                target_mentions = []
542                i = 0
543                for mention in mentions:
544                    if not isinstance(mention, Mention):
545                        __type_error__("mentions[%i]" % i, "Mention", mention)
546                    target_mentions.append(mention)
547            else:
548                __type_error__("mentions", "Optional[Iterable[Mention] | Mention]", mentions)
549        if target_mentions is not None and len(target_mentions) == 0:
550            raise ValueError("mentions cannot be empty")
551        # Validate quote:
552        if quote is not None and not isinstance(quote, Quote):
553            __type_error__("quote", "SentMessage | ReceivedMessage", quote)
554        # Validate sticker:
555        if sticker is not None:
556            if not isinstance(sticker, Sticker):
557                raise __type_error__("sticker", "Sticker", sticker)
558        # Validate preview:
559        if preview is not None:
560            if not isinstance(preview, Preview):
561                __type_error__("preview", "Preview", preview)
562            if body.find(preview.url) == -1:
563                error_message = "FATAL: error while sending message. preview URL must appear in body of message."
564                raise ValueError(error_message)
565        # Validate conflicting options:
566        if sticker is not None:
567            if body is not None or attachments is not None:
568                error_message = "If body or attachments are defined, sticker must be None."
569                raise ValueError(error_message)
570            if mentions is not None:
571                error_message = "If sticker is defined, mentions must be None"
572                raise ValueError(error_message)
573            if quote is not None:
574                error_message = "If sticker is defined, quote must be None"
575                raise ValueError(error_message)
576        # Create the send message command object:
577        send_command_obj = {
578            "jsonrpc": "2.0",
