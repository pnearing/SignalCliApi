196        """
197        Stop the signal-cli process.

199        """
200        # Close the sockets:
201        __socket_close__(self._sync_socket)

204        if self._process is not None:

206            out, err = self._process.communicate()  # Flush the pipes.
207            self._process = None  # Clear the process.
208        if self._link_process is not None:
209            self._link_process.terminate()
210            out, err = self._link_process.communicate()
211            self._link_process = None
212        # Remove socket file:
213        try:
214            if isinstance(self._server_address, str):
215                if os.path.exists(self._server_address):
216                    os.remove(self._server_address)
217        except PermissionError:
218            pass
219        return

221    def register_account(self,
222                         number: str,
223                         captcha: str,
224                         voice: bool = False
225                         ) -> tuple[bool, Account | str]:
226        """
227                Register a new account. NOTE: Subject to rate limiting.
228                :param number: str: The phone number to register.
229                :param captcha: str: The captcha from 'https://signalcaptchas.org/registration/generate.html', can
230                                        include the 'signalcaptcha://'.
231                :param voice: bool: True = Voice call verification, False = SMS verification.
232                :returns: tuple[bool, Account | str]: The first element (bool) is True for success or failure.  The
233                                                        second element on success is the new Account object, which
234                                                        will remain in an unregistered state until verify is called
235                                                        with the verification code.  Upon failure, the second element
236                                                        will contain a string with an error message.
237                :raises: TypeError: If number or captcha are not strings, or if voice is not a boolean.
238                """
239        # Type check arguments:
240        if not isinstance(number, str):
241            __type_error__("number", "str", number)
242        if not isinstance(captcha, str):
243            __type_error__("captcha", "str", captcha)
244        if not isinstance(voice, bool):
245            __type_error__("voice", "bool", voice)
246        # Value Check arguments:
247        number_match = phone_number_regex.match(number)
248        if number_match is None:
249            # noinspection SpellCheckingInspection
250            return False, "number must be in format +nnnnnnnn...."
251        # noinspection SpellCheckingInspection
252        if captcha.startswith('signalcaptcha://'):
253            captcha = captcha[16:]
254        # noinspection SpellCheckingInspection
255        if not captcha.startswith('signal-recaptcha-') and not captcha.startswith('signal-hcaptcha'):
256            return False, "invalid captcha"
257        # Check if the account exists, and isn't registered.
258        account = self.accounts.get_by_number(number)
259        if account is not None:
260            if account.registered:
261                return False, "Account already registered."
262        # Create register account command object and json command string:
263        register_account_command_obj = {
264            "jsonrpc": "2.0",
265            "id": 0,
266            "method": "register",
267            "params": {
268                "account": number,
269                "captcha": captcha,
270                "voice": voice,
271            }
272        }
273        json_command_str = json.dumps(register_account_command_obj) + '\n'
274        # Communicate with signal:
275        __socket_send__(self._sync_socket, json_command_str)
276        response_str = __socket_receive__(self._sync_socket)
277        # Parse response:
278        response_obj: dict = json.loads(response_str)
279        # Check for error:
280        if 'error' in response_obj.keys():
281            error_message = "ERROR: signal error, code: %i, message: %s" % (
282                response_obj['error']['code'], response_obj['error']['message'])
283            # Delete local account data, since signal-cli creates data for the account.
284            delete_local_data_command_obj = {
285                "jsonrpc": "2.0",
286                "id": 1,
287                "method": "deleteLocalAccountData",
288                "params": {
289                    "account": number
290                }
291            }
292            json_command_str = json.dumps(delete_local_data_command_obj) + '\n'
293            # Communicate with signal:
294            __socket_send__(self._sync_socket, json_command_str)
295            response_str = __socket_receive__(self._sync_socket)
296            # print(responseStr)
297            return False, error_message
298        # No error found, get the new account:
299        new_accounts = self.accounts.__sync__()
300        if len(new_accounts) == 0:
301            new_account = None
302        else:
303            new_account = new_accounts[0]
304        if new_account is None:
305            new_account = self.accounts.get_by_number(number)
306        return True, new_account

308    def start_link_account(self, name: Optional[str] = None) -> tuple[str, str, str]:
309        """
310        Start the link process for linking an existing account.
311        :param name: Optional[str]: The name for this device.
312        :returns: tuple[str, str, str]: The first element is the link url generated by signal, the second element is a
313                                            text QR code string, or an empty string if qrencode is not installed.  The
314                                            third element is a path to a generated png qrencode, or an empty string if
315                                            qrencode is not installed. qrencode can be installed on debian based
316                                            systems using sudo apt-get install qrencode.
317        :raises: TypeError: If name is not a string or None.
318        :raises: RuntimeError: If a link is already in progress when start_link_account is called a second time and
319                                finish_link has not been called.
320        """
321        # Type check name:
322        if name is not None and not isinstance(name, str):
323            __type_error__("name", "Optional[str]", name)
324        # Check for a running process:
325        if self._link_process is not None:
326            raise RuntimeError("link already in process.")
327        # create signal link command line:
328        link_command_line = [self._signal_exec_path, '--config', self.config_path, 'link']
329        if name is not None:
330            link_command_line.extend(('--name', name))
331        # Run the signal link process:
332        try:
333            self._link_process = Popen(link_command_line, text=True, stdout=PIPE, stderr=PIPE)
334        except CalledProcessError as e:
335            parse_signal_return_code(e.returncode, link_command_line, e.output)
336            # :NO RETURN :
337        # Gather the link from stdout
338        signal_link = self._link_process.stdout.readline()
339        signal_link = signal_link.rstrip()
340        # Generate text qrcode:
341        try:
342            text_qr_code = check_output([self._qrencode_exec_path, '-o', '-', '--type=UTF8', signal_link])
343            text_qr_code = text_qr_code.decode('UTF8')
344        except CalledProcessError:
345            text_qr_code = ''
346        # Generate png qrcode:
347        png_qr_code_file_name = "link-%s-qrcode.png" % str(name)
348        png_qr_code_path: str = os.path.join(self.config_path, png_qr_code_file_name)
349        try:
350            check_call([self._qrencode_exec_path, '-o', png_qr_code_path, signal_link])
351        except CalledProcessError:
352            png_qr_code_path = ''
353        return signal_link, text_qr_code, png_qr_code_path

355    def finish_link(self) -> tuple[bool, Account | str]:
356        """
357        Finish the linking process after confirming the link on the primary device.
358        :returns: tuple[bool, Account | str]: The first element is a bool representing success or failure; The second
359                                                element on success will be the new Account object, or on failure will be
360                                                a string containing an error message.
361        :raises: RuntimeError: In the event that signal-cli success response changes.
362        """
363        # Check for a link process:
364        if self._link_process is None:
365            error_message = "link not started"
366            return False, error_message
367        # Wait for the process:
368        return_code = self._link_process.wait()
369        # Parse return_code:
370        if return_code == 0:
371            new_accounts = self.accounts.__sync__()
372            if len(new_accounts) != 0:
373                new_account = new_accounts[0]
374            else:
375                new_account = None
376            response_line = self._link_process.stdout.readline()
377            if DEBUG:
378                print(response_line)
379            if new_account is None:
380                regex = r'^Associated with: (?P<number>.*)$'
381                match = re.match(regex, response_line)
382                if match is not None:
383                    new_account = self.accounts.get_by_number(match['number'])
384                    return True, new_account
385                else:
386                    # Shouldn't get here, return code is 0 (success).
387                    raise RuntimeError("success response from signal must have changed.")
388        elif return_code == 1:
389            response_string = self._link_process.stderr.read()
390            regex = r'The user (?P<number>\+\d+) already exists'
391            match = re.match(regex, response_string)
392            if match is not None:
393                error_message = "Account '%s' already exists." % match["number"]
394                return False, error_message
395        elif return_code == 3:
396            response_string = self._link_process.stderr.read()
397            regex = r'Link request error: Connection closed!'
398            match = re.match(regex, response_string)
399            if match is not None:
400                error_message = "Link request timeout."
401                return False, error_message
402        response_string = self._link_process.stderr.read()
403        return False, response_string
