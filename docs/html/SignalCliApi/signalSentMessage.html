405    def __parse_receipt__(self, receipt: Receipt) -> None:
406        if receipt.receiptType == Receipt.TYPE_DELIVERY:
407            self.mark_delivered(receipt.when)
408            self.delivery_receipts.append(receipt)
409        elif receipt.receiptType == Receipt.TYPE_READ:
410            self.mark_read(receipt.when)
411            self.read_receipts.append(receipt)
412        elif receipt.receiptType == Receipt.TYPE_VIEWED:
413            self.mark_viewed(receipt.when)
414            self.viewed_receipts.append(receipt)
415        else:
416            errorMessage = "FATAL: Invalid receipt type, cannot parse. SentMessage.__parse_receipt__"
417            raise RuntimeError(errorMessage)
